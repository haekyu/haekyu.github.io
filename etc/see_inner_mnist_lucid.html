# Make a network whose features we'll visualize
def make_network():
    x = tf.placeholder(tf.float32, [None, 28, 28, 3], name='input')
    y = tf.placeholder(tf.int64, [None])
    training = tf.Variable(True, name='training')

    conv1 = tf.layers.conv2d(
        inputs=x,
        filters=32,
        kernel_size=5,
        padding="same",
        activation=tf.nn.relu)
    
    pool1 = tf.layers.max_pooling2d(
      inputs=conv1,
      pool_size=[2, 2],
      strides=2)
    
    conv2 = tf.layers.conv2d(
      inputs=pool1,
      filters=64,
      kernel_size=[5, 5],
      padding="same",
      activation=tf.nn.relu)
    
    pool2 = tf.layers.max_pooling2d(
      inputs=conv2,
      pool_size=[2, 2],
      strides=2)

    pool2_flat = tf.reshape(pool2, [-1, 7 * 7 * 64])
    
    fc = tf.layers.dense(
        inputs=pool2_flat, 
        units=1024, 
        activation=tf.nn.relu)

    logits = tf.layers.dense(inputs=fc, units=10)
    
    # Identity the layers to make them into the graph..?
    tf.identity(conv1, 'conv1')
    tf.identity(pool1, 'pool1')
    tf.identity(conv2, 'conv2')
    tf.identity(pool2, 'pool2')
    tf.identity(fc, 'fc')
    tf.identity(logits, 'logits')
    
    # Layers
    layers = [x, y, training]
    layers += [conv1, pool1, conv2, pool2, fc, logits]
    
    return layers

# Make train optimizer
def make_train_op(logits, y):
    # Make prediction
    correct_prediction = tf.equal(tf.argmax(logits, axis=1), y)
    
    # Measure accuracy
    accuracy_tensor = tf.reduce_mean(tf.cast(correct_prediction, tf.float32))
    loss_tensor = tf.reduce_mean(tf.losses.sparse_softmax_cross_entropy(labels=y, logits=logits))
    
    # Make training optimizer
    optimizer = tf.train.AdamOptimizer()
    train_op = optimizer.minimize(loss_tensor, tf.train.get_or_create_global_step())
    
    return train_op, loss_tensor, accuracy_tensor

# Train the model
def train(sess, train_data, train_op, loss_tensor, accuracy_tensor, x, y, n_batches):
    for _ in range(n_batches):
        # Get batch
        x_batch, y_batch = train_data.next_batch(128)
        x_batch = x_batch.reshape([-1, 28, 28, 1])
        x_batch = np.tile(x_batch, 3)
        
        # Train
        sess.run(train_op, feed_dict={x: x_batch, y: y_batch})

# Save a frozen graph, which includes both the graph structure and weights
def save_frozen_graph(filename):
    output_graph_def = tf.graph_util.convert_variables_to_constants(
      sess,
      tf.get_default_graph().as_graph_def(),
      ['conv1', 'pool1', 'conv2', 'pool2', 'fc', 'logits']
    )
    with open(filename, "wb") as f:
        f.write(output_graph_def.SerializeToString())

# Render the visualization
def render_vis(model, objective_f, param_f, steps=512):
    T = render.make_vis_T(model, objective_f, param_f, optimizer=None, transforms=[])
    loss, vis_op, t_image = T("loss"), T("vis_op"), T("input")
    sess = tf.Session()
    sess.run(tf.global_variables_initializer())
    for step in range(steps):
        sess.run(vis_op)
    return sess.run(t_image)
